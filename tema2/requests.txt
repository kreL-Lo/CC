

API FOR QUESTIONS/ANSWER LIKE REDDIT

PUT /THREAD/{:NEW_THREAD}
PUT /THREAD/{:ID}/QUESTION
PUT /THREAD/{:ID}/QUESTION/{:ID}/ANSWER

POST /THREAD/{:ID}/QUESTION
POST /THREAD/{:ID}/QUESTION/{:ID}/ANSWER


GET /THREADS
GET /THREAD/{:ID}/QUESTIONS
GET /THREAD/{:ID}/QUESTION/{:ID}
GET /THREAD/{:ID}/QUESTION/{:ID}/ANSWER/{:ID}

DELETE /THREAD/{ID}
DELETE /QUESTION/{ID}
DELETE /ANSWER /{ID}
